\begin{frame}[t]{ここまでのまとめ}{branch を使用する}

  \begin{itemize}
  \item git branch は branch の一覧、作成、削除
    \vspace{2ex}

  \item git checkout は HEAD の切り替え

    会わせて、workspace, cached も変更
    \vspace{2ex}

  \item git merge は他の branch の変更を取り入れる
    \vspace{2ex}

  \item branch の作成、削除で commit が

    作成、削除される事は無い
  \end{itemize}

\end{frame}
