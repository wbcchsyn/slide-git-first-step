\begin{frame}[t]{ここまでのまとめ}{その他}

  \begin{itemize}
  \item git revert は過去の commit を打ち消す commit 作成
    \vspace{2ex}

  \item git tag は tag 一覧表示、作成、削除
    \vspace{2ex}

  \item branch や HEAD は移動するが tag は移動しないので、リリース管理等に使用可能
  \end{itemize}

\end{frame}
